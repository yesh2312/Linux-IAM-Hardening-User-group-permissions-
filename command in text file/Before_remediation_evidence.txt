Finding 3: World-writable Log Directory

Before Remediation Evidence

echo "=== BEFORE: World-writable Log Directory ==="
ls -la /var/log/ | grep insecure-app
ls -la /var/log/insecure-app/

Remediation Steps

# Secure log directory and files
sudo chmod 755 /var/log/insecure-app
sudo chmod 644 /var/log/insecure-app/application.log
sudo chown root:root /var/log/insecure-app/ /var/log/insecure-app/application.log

# Alternatively, remove if not needed:
# sudo rm -rf /var/log/insecure-app/

After Remediation Evidence

echo "=== AFTER: Secure Log Directory ==="
ls -la /var/log/ | grep insecure-app
ls -la /var/log/insecure-app/
